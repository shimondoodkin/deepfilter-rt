1//! # DeepFilter-RT
2//!
3//! Real-time noise suppression using DeepFilterNet neural network with ONNX Runtime inference.
4//!
5//! This crate provides frame-by-frame audio enhancement for real-time applications such as
6//! voice calls, streaming, and live audio processing. It implements the two-stage DeepFilterNet
7//! algorithm: ERB-based spectral masking followed by deep filtering on low frequencies.
8//!
9//! ## Model Variants
10//!
11//! | Variant | Folder | Latency | Mode | Use Case |
12//! |---------|--------|---------|------|----------|
13//! | DeepFilterNet2 | `dfn2` | 30ms | Stateless | General purpose |
14//! | DeepFilterNet2-LL | `dfn2_ll` | 10ms | Stateless | Low-latency |
15//! | DeepFilterNet2-H0 | `dfn2_h0` | 30ms | Stateful | Best quality (GRU) |
16//! | DeepFilterNet3 | `dfn3` | 30ms | Stateless | High quality |
17//! | DeepFilterNet3-LL | `dfn3_ll` | 10ms | Stateless | Real-time |
18//! | DeepFilterNet3-H0 | `dfn3_h0` | 30ms | Stateful | Best quality (GRU) |
19//!
20//! ## Audio Parameters
21//!
22//! - **Sample rate**: 48 kHz (resample if needed)
23//! - **Frame size**: 480 samples (10 ms)
24//! - **FFT size**: 960 samples
25//! - **Frequency bins**: 481 (FFT_SIZE/2 + 1)
26//!
27//! ## Quick Start
28//!
29//! ```ignore
30//! use deepfilter_rt::{DeepFilterProcessor, HOP_SIZE};
31//! use std::path::Path;
32//!
33//! let mut processor = DeepFilterProcessor::new(Path::new("models/dfn3"))?;
34//! processor.warmup()?; // Avoid cold-start latency
35//!
36//! // Process 10ms frames
37//! let mut output = vec![0.0f32; HOP_SIZE];
38//! processor.process_frame(&input_samples, &mut output)?;
39//! ```
40//!
41//! ## Constructors
42//!
43//! ### DeepFilterProcessor
44//!
45//! - [`DeepFilterProcessor::new`] - Auto-detect mode from folder name
46//! - [`DeepFilterProcessor::new_with_mode`] - Explicit [`InferenceMode`]
47//! - [`DeepFilterProcessor::with_variant`] - Explicit [`ModelVariant`]
48//! - [`DeepFilterProcessor::with_variant_and_threads`] - With thread count
49//! - [`DeepFilterProcessor::with_variant_mode_and_threads`] - Full control
50//!
51//! ### DeepFilterStream
52//!
53//! - [`DeepFilterStream::new`] - Auto-detect mode from folder name
54//! - [`DeepFilterStream::new_with_mode`] - Explicit [`InferenceMode`]
55//! - [`DeepFilterStream::with_threads`] - With thread count
56//!
57//! ## Inference Modes
58//!
59//! [`InferenceMode`] determines how temporal context is handled:
60//!
61//! - **`StatefulH0`** - Uses GRU hidden states (`h0`/`h1`). Requires `*_h0` models.
62//!   Best quality, maintains context across frames.
63//! - **`StatelessWindowLast`** - Uses sliding window with lookahead.
64//!   Works with all models, no persistent state.
65//!
66//! Mode is auto-detected from folder name (`*_h0` → Stateful, otherwise → Stateless).
67//!
68//! ## Thread Safety
69//!
70//! Each [`DeepFilterProcessor`] instance is independent and can run on its own thread.
71//! Create separate instances for parallel processing - they do not share state.
72

84// Global ORT initialization guard - ensures init is called only once
85static ORT_INIT: Once = Once::new();
86static mut ORT_INIT_ERROR: Option<String> = None;
87
88fn init_ort() -> Result<()> {
89    ORT_INIT.call_once(|| {
90        // With load-dynamic feature, we need to specify where to find the library
91        #[cfg(target_os = "android")]
92        let lib_name = "libonnxruntime.so";
93
94        #[cfg(target_os = "windows")]
95        let lib_name = "onnxruntime";
96
97        #[cfg(all(not(target_os = "android"), not(target_os = "windows")))]
98        let lib_name = "libonnxruntime";
99
100        match ort::init_from(lib_name) {
101            Ok(builder) => {
102                let _ = builder.with_name("deepfilter").commit();
103            }
104            Err(e) => unsafe {
105                ORT_INIT_ERROR = Some(format!("Failed to load ONNX Runtime from '{}': {}", lib_name, e));
106            }
107        }
108    });
109
110    // Check if initialization failed
111    unsafe {
112        if let Some(ref err) = ORT_INIT_ERROR {
113            return Err(DfError::Config(err.clone()));
114        }
115    }
116    Ok(())
117}

120pub const SAMPLE_RATE: usize = 48000;
121pub const FFT_SIZE: usize = 960;
122pub const HOP_SIZE: usize = 480;
123pub const FREQ_SIZE: usize = 481;

129pub const MIN_DB_THRESH: f32 = -15.0;

133/// Runtime inference mode override.
134#[derive(Debug, Clone, Copy, PartialEq)]
135pub enum InferenceMode {
136    StatefulH0,
137    StatelessWindowLast,
138}
139
140impl InferenceMode {
141    /// Resolve mode from the model directory name (`*_h0` -> stateful).
142    fn from_model_dir(model_dir: &Path) -> Self {
143        let name = model_dir
144            .file_name()
145            .and_then(|n| n.to_str())
146            .unwrap_or_default()
147            .to_ascii_lowercase();
148        if name.ends_with("_h0") {
149            InferenceMode::StatefulH0
150        } else {
151            InferenceMode::StatelessWindowLast
152        }
153    }
154
155    /// Minimum frames needed before inference can run.
156    fn min_required_frames(self, lookahead: usize) -> usize {
157        match self {
158            InferenceMode::StatefulH0 => 1,
159            InferenceMode::StatelessWindowLast => lookahead + 1,
160        }
161    }
162}

178/// Model variant detection from config.ini
179#[derive(Debug, Clone, Copy, PartialEq)]
180pub enum ModelVariant {
181    DeepFilterNet2,
182    DeepFilterNet2LL,
183    DeepFilterNet3,
184    DeepFilterNet3LL,
185}
186
187impl ModelVariant {
188    /// Detect variant from config.ini in model directory
189    pub fn from_config(model_dir: &Path) -> Result<Self> {
190        let config_path = model_dir.join("config.ini");
191        let content = fs::read_to_string(&config_path)
192            .map_err(|e| DfError::Config(format!("Failed to read config.ini: {}", e)))?;
193
194        let params = parse_ini(&content);
195
196        let model = params.get("model").map(|s| s.as_str()).unwrap_or("deepfilternet2");
197        let lookahead = params.get("df_lookahead")
198            .and_then(|s| s.parse::<i32>().ok())
199            .unwrap_or(2);
200
201        match (model, lookahead) {
202            ("deepfilternet2", 0) => Ok(ModelVariant::DeepFilterNet2LL),
203            ("deepfilternet2", _) => Ok(ModelVariant::DeepFilterNet2),
204            ("deepfilternet3", 0) => Ok(ModelVariant::DeepFilterNet3LL),
205            ("deepfilternet3", _) => Ok(ModelVariant::DeepFilterNet3),
206            _ => Ok(ModelVariant::DeepFilterNet3), // Default
207        }
208    }
209
210    pub fn is_low_latency(&self) -> bool {
211        matches!(self, ModelVariant::DeepFilterNet2LL | ModelVariant::DeepFilterNet3LL)
212    }
213
214    pub fn name(&self) -> &'static str {
215        match self {
216            ModelVariant::DeepFilterNet2 => "DeepFilterNet2",
217            ModelVariant::DeepFilterNet2LL => "DeepFilterNet2-LL",
218            ModelVariant::DeepFilterNet3 => "DeepFilterNet3",
219            ModelVariant::DeepFilterNet3LL => "DeepFilterNet3-LL",
220        }
221    }
222}

224/// Simple INI parser
225fn parse_ini(content: &str) -> HashMap<String, String> {
226    let mut map = HashMap::new();
227    for line in content.lines() {
228        let line = line.trim();
229        if line.starts_with('[') || line.starts_with('#') || line.is_empty() {
230            continue;
231        }
232        if let Some((key, value)) = line.split_once('=') {
233            map.insert(key.trim().to_string(), value.trim().to_string());
234        }
235    }
236    map
237}

300    /// Create processor with explicit variant, mode, and thread count.
301    pub fn with_variant_mode_and_threads(
302        model_dir: &Path,
303        variant: ModelVariant,
304        inference_mode: InferenceMode,
305        intra_threads: Option<usize>,
306    ) -> Result<Self> {
307        init_ort()?;
308
309        let (df_lookahead, conv_lookahead, min_nb_erb_freqs, norm_alpha, enc_kernel_t, enc_hidden_dim) = {
310            let config_path = model_dir.join("config.ini");
311            let content = fs::read_to_string(&config_path).unwrap_or_default();
312            let params = parse_ini(&content);
313            let df_lookahead = params
314                .get("df_lookahead")
315                .and_then(|s| s.parse::<usize>().ok())
316                .unwrap_or(2);
317            let conv_lookahead = params
318                .get("conv_lookahead")
319                .and_then(|s| s.parse::<usize>().ok())
320                .unwrap_or(0);
321            let min_nb_erb_freqs = params
322                .get("min_nb_erb_freqs")
323                .and_then(|s| s.parse::<usize>().ok())
324                .unwrap_or(1);
325            let enc_kernel_t = params
326                .get("conv_kernel_inp")
327                .and_then(|s| s.split(',').next())
328                .and_then(|s| s.trim().parse::<usize>().ok())
329                .unwrap_or(3);
330            let enc_hidden_dim = params
331                .get("emb_hidden_dim")
332                .and_then(|s| s.parse::<usize>().ok())
333                .unwrap_or(256);
334            let norm_alpha = if let Some(tau) = params
335                .get("norm_tau")
336                .and_then(|s| s.parse::<f32>().ok())
337            {
338                let dt = HOP_SIZE as f32 / SAMPLE_RATE as f32;
339                let a = f32::exp(-dt / tau);
340                // Match python get_norm_alpha rounding behavior
341                let mut precision: u32 = 3;
342                let mut rounded = 1.0f32;
343                while rounded >= 1.0 {
344                    let scale = 10f32.powi(precision as i32);
345                    rounded = (a * scale).round() / scale;
346                    precision += 1;
347                }
348                rounded
349            } else {
350                DEFAULT_NORM_ALPHA
351            };
352            (df_lookahead, conv_lookahead, min_nb_erb_freqs, norm_alpha, enc_kernel_t, enc_hidden_dim)
353        };
354        let lookahead = df_lookahead.max(conv_lookahead);
355
356        let build_session = |path: std::path::PathBuf| -> Result<Session> {
357            let mut builder = Session::builder()?
358                .with_optimization_level(GraphOptimizationLevel::Level3)?;
359
360            if let Some(threads) = intra_threads {
361                builder = builder.with_intra_threads(threads)?;
362            }
363
364            Ok(builder.commit_from_file(path)?)
365        };
366
367        let encoder = build_session(model_dir.join("enc.onnx"))?;
368        let erb_decoder = build_session(model_dir.join("erb_dec.onnx"))?;
369        let df_decoder = build_session(model_dir.join("df_dec.onnx"))?;
370        let has_h0 = encoder.inputs().iter().any(|i| i.name() == "h0");
371        if inference_mode == InferenceMode::StatefulH0 && !has_h0 {
372            return Err(DfError::Config(
373                "Model directory ends with _h0 but encoder has no h0 input".to_string(),
374            ));
375        }
376        let enc_window = match inference_mode {
377            InferenceMode::StatefulH0 => enc_kernel_t,
378            InferenceMode::StatelessWindowLast => enc_kernel_t + 2 * lookahead.max(1),
379        };
380
381        let mut df_state = DFState::new(SAMPLE_RATE, FFT_SIZE, HOP_SIZE, NB_ERB, min_nb_erb_freqs);
382        df_state.init_norm_states(NB_DF);
383
384        let buf_x_len = DF_ORDER.max(lookahead.max(1));
385        let mut rolling_spec_buf_x = VecDeque::with_capacity(buf_x_len);
386        for _ in 0..buf_x_len {
387            rolling_spec_buf_x.push_back(vec![Complex32::new(0.0, 0.0); FREQ_SIZE]);
388        }
389        let mut rolling_spec_buf_y = VecDeque::with_capacity(DF_ORDER + lookahead);
390        for _ in 0..(DF_ORDER + lookahead) {
391            rolling_spec_buf_y.push_back(vec![Complex32::new(0.0, 0.0); FREQ_SIZE]);
392        }
393        let erb_feat_buf = VecDeque::with_capacity(lookahead + 1);
394        let spec_feat_buf = VecDeque::with_capacity(lookahead + 1);
395        let enc_feat_buf_erb = VecDeque::with_capacity(enc_window);
396        let enc_feat_buf_spec = VecDeque::with_capacity(enc_window);
397        let enc_h = vec![0.0f32; enc_hidden_dim];
398
399        let debug_dump_frame = std::env::var("DF_DEBUG_DUMP_FRAME")
400            .ok()
401            .and_then(|v| v.parse::<usize>().ok());
402
403        Ok(Self {
404            encoder,
405            erb_decoder,
406            df_decoder,
407            df_state,
408            rolling_spec_buf_x,
409            rolling_spec_buf_y,
410            erb_feat_buf,
411            spec_feat_buf,
412            enc_feat_buf_erb,
413            enc_feat_buf_spec,
414            frames_processed: 0,
415            lookahead,
416            enc_h,
417            enc_hidden_dim,
418            enc_window,
419            inference_mode,
420            norm_alpha,
421            variant,
422            debug_dump_frame,
423        })
424    }

430    pub fn reset(&mut self) {
431        self.df_state.reset();
432        self.df_state.init_norm_states(NB_DF);
433        self.frames_processed = 0;
434        // Keep debug_dump_frame across resets
435        self.rolling_spec_buf_x.clear();
436        let buf_x_len = DF_ORDER.max(self.lookahead.max(1));
437        for _ in 0..buf_x_len {
438            self.rolling_spec_buf_x
439                .push_back(vec![Complex32::new(0.0, 0.0); FREQ_SIZE]);
440        }
441        self.rolling_spec_buf_y.clear();
442        for _ in 0..(DF_ORDER + self.lookahead) {
443            self.rolling_spec_buf_y
444                .push_back(vec![Complex32::new(0.0, 0.0); FREQ_SIZE]);
445        }
446        self.erb_feat_buf.clear();
447        self.spec_feat_buf.clear();
448        self.enc_feat_buf_erb.clear();
449        self.enc_feat_buf_spec.clear();
450        self.enc_h.fill(0.0);
451    }

453    /// Perform warm-up inference to avoid cold-start latency
454    ///
455    /// NNAPI and other execution providers often have significant first-inference
456    /// latency (100ms+). Call this during initialization to "wake up" the GPU/NPU.
457    /// The processor state is reset after warmup.
458    pub fn warmup(&mut self) -> Result<()> {
459        let dummy_input = vec![0.0f32; HOP_SIZE];
460        let mut dummy_output = vec![0.0f32; HOP_SIZE];
461
462        // Run a few frames to fully warm up the pipeline
463        for _ in 0..3 {
464            self.process_frame(&dummy_input, &mut dummy_output)?;
465        }
466
467        // Reset state so warmup doesn't affect real processing
468        self.reset();
469        Ok(())
470    }

472    /// Process single frame (480 samples @ 48kHz = 10ms)
473    pub fn process_frame(&mut self, input: &[f32], output: &mut [f32]) -> Result<()> {
474        debug_assert_eq!(input.len(), HOP_SIZE);
475        debug_assert_eq!(output.len(), HOP_SIZE);
476
477        // 1. STFT
478        let mut spec = vec![Complex32::new(0.0, 0.0); FREQ_SIZE];
479        self.df_state.analysis(input, &mut spec);
480
481        // 2. Store ORIGINAL spectrum in rolling buffers BEFORE any processing (for DF)
482        self.rolling_spec_buf_x.pop_front();
483        self.rolling_spec_buf_x.push_back(spec.clone());
484        self.rolling_spec_buf_y.pop_front();
485        self.rolling_spec_buf_y.push_back(spec.clone());
486
487        // 3. ERB features
488        let mut erb_feat = vec![0.0f32; NB_ERB];
489        self.df_state.feat_erb(&spec, self.norm_alpha, &mut erb_feat);
490
491        // 4. Spec features - use separate input/output buffers
492        let spec_feat_input: Vec<Complex32> = spec[..NB_DF].to_vec();
493        let mut spec_feat_cplx = vec![Complex32::new(0.0, 0.0); NB_DF];
494        self.df_state
495            .feat_cplx(&spec_feat_input, self.norm_alpha, &mut spec_feat_cplx);
496
497        // 5. Buffer features for lookahead alignment (pad_feat behavior)
498        self.erb_feat_buf.push_back(erb_feat.clone());
499        self.spec_feat_buf.push_back(spec_feat_cplx.clone());
500        if self.erb_feat_buf.len() <= self.lookahead {
501            // Not enough lookahead frames yet
502            self.frames_processed = self.frames_processed.saturating_add(1);
503            let mut out_spec = vec![Complex32::new(0.0, 0.0); FREQ_SIZE];
504            self.df_state.synthesis(&mut out_spec, output);
505            return Ok(());
506        }
507        // Use lookahead frame for encoder input and build temporal context
508        let inf_erb = self.erb_feat_buf[self.lookahead].clone();
509        let inf_spec = self.spec_feat_buf[self.lookahead].clone();
510        // Shift buffer: remove oldest once consumed for lookahead
511        if self.erb_feat_buf.len() > self.lookahead + 1 {
512            self.erb_feat_buf.pop_front();
513        }
514        if self.spec_feat_buf.len() > self.lookahead + 1 {
515            self.spec_feat_buf.pop_front();
516        }
517        self.enc_feat_buf_erb.push_back(inf_erb);
518        self.enc_feat_buf_spec.push_back(inf_spec);
519        let min_required = self.inference_mode.min_required_frames(self.lookahead);
520        if self.enc_feat_buf_erb.len() < min_required {
521            // Not enough temporal context yet
522            self.frames_processed = self.frames_processed.saturating_add(1);
523            let mut out_spec = vec![Complex32::new(0.0, 0.0); FREQ_SIZE];
524            self.df_state.synthesis(&mut out_spec, output);
525            return Ok(());
526        }
527        if self.enc_feat_buf_erb.len() > self.enc_window {
528            self.enc_feat_buf_erb.pop_front();
529            self.enc_feat_buf_spec.pop_front();
530        }
531        let enc_erb_seq: Vec<Vec<f32>> = self.enc_feat_buf_erb.iter().cloned().collect();
532        let enc_spec_seq: Vec<Vec<Complex32>> = self.enc_feat_buf_spec.iter().cloned().collect();
533        let take_idx = enc_erb_seq.len().saturating_sub(1);
534        let (mask, df_coefs, lsnr) =
535            self.run_inference_seq(&enc_erb_seq, &enc_spec_seq, take_idx)?;
536
537        if let Some(target) = self.debug_dump_frame {
538            if self.frames_processed == target {
539                self.dump_debug_frame(target, &erb_feat, &spec_feat_cplx, &mask, &df_coefs)?;
540            }
541        }
542
543        // 6. Apply ERB mask to the centered frame in the rolling buffer (df_order - 1)
544        let apply_mask = lsnr <= MAX_DB_ERB_THRESH;
545        let apply_zero_mask = lsnr < MIN_DB_THRESH;
546        if let Some(center_spec) = self.rolling_spec_buf_y.get_mut(DF_ORDER - 1) {
547            if apply_mask {
548                if apply_zero_mask {
549                    let zeros = vec![0.0f32; NB_ERB];
550                    self.df_state.apply_mask(center_spec, &zeros);
551                } else {
552                    self.df_state.apply_mask(center_spec, &mask);
553                }
554            }
555        }
556
557        // 7. Output the centered frame (implicit lookahead via rolling buffer)
558        self.frames_processed = self.frames_processed.saturating_add(1);
559        let mut out_spec = self
560            .rolling_spec_buf_y
561            .get(DF_ORDER - 1)
562            .cloned()
563            .unwrap_or_else(|| vec![Complex32::new(0.0, 0.0); FREQ_SIZE]);
564
565        let apply_df = lsnr <= MAX_DB_DF_THRESH && !apply_zero_mask;
566        if apply_df && self.rolling_spec_buf_x.len() >= DF_ORDER {
567            self.apply_deep_filter(&mut out_spec, &df_coefs);
568        }
569
570        // 8. ISTFT
571        self.df_state.synthesis(&mut out_spec, output);
572
573        Ok(())
574    }
575
576    fn run_inference_seq(
577        &mut self,
578        erb_feat_seq: &[Vec<f32>],
579        spec_feat_seq: &[Vec<Complex32>],
580        take_idx: usize,
581    ) -> Result<(Vec<f32>, Array3<f32>, f32)> {
582        let t = erb_feat_seq.len();
583        debug_assert_eq!(t, spec_feat_seq.len());
584        // ERB tensor: [1, 1, T, 32]
585        let mut erb_data = vec![0.0f32; t * NB_ERB];
586        for (ti, frame) in erb_feat_seq.iter().enumerate() {
587            let base = ti * NB_ERB;
588            erb_data[base..base + NB_ERB].copy_from_slice(frame);
589        }
590        let erb_tensor = Tensor::from_array(([1usize, 1, t, NB_ERB], erb_data))?;
591
592        // Spec tensor: [1, 2, T, 96] - channels = [real, imag]
593        let mut spec_data = vec![0.0f32; 2 * t * NB_DF];
594        // channel 0: real
595        for (ti, frame) in spec_feat_seq.iter().enumerate() {
596            let base = ti * NB_DF;
597            for (fi, c) in frame.iter().enumerate() {
598                spec_data[base + fi] = c.re;
599            }
600        }
601        // channel 1: imag
602        let ch1_offset = t * NB_DF;
603        for (ti, frame) in spec_feat_seq.iter().enumerate() {
604            let base = ch1_offset + ti * NB_DF;
605            for (fi, c) in frame.iter().enumerate() {
606                spec_data[base + fi] = c.im;
607            }
608        }
609        let spec_tensor = Tensor::from_array(([1usize, 2, t, NB_DF], spec_data))?;
610
611        // Encoder (optionally stateful)
612        let enc_outputs = if self.inference_mode == InferenceMode::StatefulH0 {
613            let h0_tensor = Tensor::from_array(([1usize, 1, self.enc_hidden_dim], self.enc_h.clone()))?;
614            self.encoder.run(ort::inputs![
615                "feat_erb" => erb_tensor,
616                "feat_spec" => spec_tensor,
617                "h0" => h0_tensor,
618            ])?
619        } else {
620            self.encoder.run(ort::inputs![
621                "feat_erb" => erb_tensor,
622                "feat_spec" => spec_tensor,
623            ])?
624        };
625        // Extract encoder outputs as (shape, data) tuples
626        let (e0_shape, e0_data) = enc_outputs["e0"].try_extract_tensor::<f32>()?;
627        let (e1_shape, e1_data) = enc_outputs["e1"].try_extract_tensor::<f32>()?;
628        let (e2_shape, e2_data) = enc_outputs["e2"].try_extract_tensor::<f32>()?;
629        let (e3_shape, e3_data) = enc_outputs["e3"].try_extract_tensor::<f32>()?;
630        let (emb_shape, emb_data) = enc_outputs["emb"].try_extract_tensor::<f32>()?;
631        let (c0_shape, c0_data) = enc_outputs["c0"].try_extract_tensor::<f32>()?;
632        let (_lsnr_shape, lsnr_data) = enc_outputs["lsnr"].try_extract_tensor::<f32>()?;
633
634        let e0_vec = e0_data.to_vec();
635        let e1_vec = e1_data.to_vec();
636        let e2_vec = e2_data.to_vec();
637        let e3_vec = e3_data.to_vec();
638        let emb_vec: Vec<f32> = emb_data.to_vec();
639        let c0_vec: Vec<f32> = c0_data.to_vec();
640        let lsnr_vec: Vec<f32> = lsnr_data.to_vec();
641        let lsnr = lsnr_vec.get(take_idx).copied().unwrap_or(0.0);
642        if self.inference_mode == InferenceMode::StatefulH0 {
643            let (_h1_shape, h1_data) = enc_outputs["h1"].try_extract_tensor::<f32>()?;
644            let h1_vec = h1_data.to_vec();
645            if h1_vec.len() >= self.enc_hidden_dim {
646                self.enc_h.copy_from_slice(&h1_vec[..self.enc_hidden_dim]);
647            }
648        }
649
650        // Helper to convert shape from [i64] to Vec<usize>
651        let to_shape = |s: &ort::tensor::Shape| -> Vec<usize> {
652            s.iter().map(|&d| d as usize).collect()
653        };
654
655        // Convert to tensors for ERB decoder
656        let e0_tensor = Tensor::from_array((to_shape(e0_shape), e0_data.to_vec()))?;
657        let e1_tensor = Tensor::from_array((to_shape(e1_shape), e1_data.to_vec()))?;
658        let e2_tensor = Tensor::from_array((to_shape(e2_shape), e2_data.to_vec()))?;
659        let e3_tensor = Tensor::from_array((to_shape(e3_shape), e3_data.to_vec()))?;
660        let e0_shape_vec = to_shape(e0_shape);
661        let e1_shape_vec = to_shape(e1_shape);
662        let e2_shape_vec = to_shape(e2_shape);
663        let e3_shape_vec = to_shape(e3_shape);
664        let emb_shape_vec = to_shape(emb_shape);
665        let c0_shape_vec = to_shape(c0_shape);
666        let emb_tensor = Tensor::from_array((emb_shape_vec.clone(), emb_vec.clone()))?;
667        let c0_tensor = Tensor::from_array((c0_shape_vec.clone(), c0_vec.clone()))?;
668
669        // Drop outputs to release mutable borrow before debug dumping
670        drop(enc_outputs);
671
672        if let Some(target) = self.debug_dump_frame {
673            if self.frames_processed == target {
674                self.dump_debug_tensor("enc_e0", target, e0_shape_vec, &e0_vec)?;
675                self.dump_debug_tensor("enc_e1", target, e1_shape_vec, &e1_vec)?;
676                self.dump_debug_tensor("enc_e2", target, e2_shape_vec, &e2_vec)?;
677                self.dump_debug_tensor("enc_e3", target, e3_shape_vec, &e3_vec)?;
678                self.dump_debug_tensor("enc_emb", target, emb_shape_vec.clone(), &emb_vec)?;
679                self.dump_debug_tensor("enc_c0", target, c0_shape_vec, &c0_vec)?;
680                self.dump_debug_tensor("enc_lsnr", target, vec![1], &lsnr_vec)?;
681            }
682        }
683
684        // ERB decoder
685        let erb_dec_outputs = self.erb_decoder.run(ort::inputs![
686            "emb" => emb_tensor,
687            "e3" => e3_tensor,
688            "e2" => e2_tensor,
689            "e1" => e1_tensor,
690            "e0" => e0_tensor,
691        ])?;
692        let (mask_shape, mask_data) = erb_dec_outputs["m"].try_extract_tensor::<f32>()?;
693        let mask_flat: Vec<f32> = mask_data.to_vec();
694        // mask shape: [1,1,T,NB_ERB]
695        let mask_t = mask_shape[2] as usize;
696        let mask_offset = take_idx.min(mask_t.saturating_sub(1)) * NB_ERB;
697        let mask: Vec<f32> = mask_flat[mask_offset..mask_offset + NB_ERB].to_vec();
698
699        // DF decoder - need fresh tensors
700        let emb_tensor2 = Tensor::from_array((emb_shape_vec, emb_vec))?;
701        let df_dec_outputs = self.df_decoder.run(ort::inputs![
702            "emb" => emb_tensor2,
703            "c0" => c0_tensor,
704        ])?;
705
706        let (coefs_shape, coefs_data) = df_dec_outputs["coefs"].try_extract_tensor::<f32>()?;
707        let coefs_flat: Vec<f32> = coefs_data.to_vec();
708        let coefs_t = coefs_shape[1] as usize;
709        let coefs_frame_size = NB_DF * DF_ORDER * 2;
710        let coefs_offset = take_idx.min(coefs_t.saturating_sub(1)) * coefs_frame_size;
711        let coefs_slice = &coefs_flat[coefs_offset..coefs_offset + coefs_frame_size];
712        // Model outputs shape [1, 1, 96, 10] = [batch, time, nb_df, df_order*2]
713        // Reshape to [nb_df, df_order, 2] for our indexing
714        let df_coefs = Array3::from_shape_vec(
715            (NB_DF, DF_ORDER, 2),
716            Self::reshape_df_coefs(coefs_slice),
717        )?;
718
719        Ok((mask, df_coefs, lsnr))
720    }
721
722    fn reshape_df_coefs(coefs: &[f32]) -> Vec<f32> {
723        // Input layout: [1, 1, 96, 10] flattened = [freq][order*2]
724        // where for each freq: [re0, im0, re1, im1, re2, im2, re3, im3, re4, im4]
725        // Output layout: [nb_df, df_order, 2] = [freq][order][re/im]
726        coefs.to_vec()
727    }

729    fn dump_debug_frame(
730        &self,
731        frame_idx: usize,
732        erb_feat: &[f32],
733        spec_feat: &[Complex32],
734        mask: &[f32],
735        df_coefs: &Array3<f32>,
736    ) -> Result<()> {
737        let out_dir = std::path::Path::new(env!("CARGO_MANIFEST_DIR"))
738            .join("debug_dump");
739        std::fs::create_dir_all(&out_dir)?;
740
741        let write_f32 = |name: &str, data: &[f32]| -> Result<()> {
742            let path = out_dir.join(format!("{}_frame_{}.txt", name, frame_idx));
743            let mut s = String::new();
744            for (i, v) in data.iter().enumerate() {
745                s.push_str(&format!("{},{}\n", i, v));
746            }
747            std::fs::write(path, s)?;
748            Ok(())
749        };
750
751        let write_cplx = |name: &str, data: &[Complex32]| -> Result<()> {
752            let path = out_dir.join(format!("{}_frame_{}.txt", name, frame_idx));
753            let mut s = String::new();
754            for (i, v) in data.iter().enumerate() {
755                s.push_str(&format!("{},{},{}\n", i, v.re, v.im));
756            }
757            std::fs::write(path, s)?;
758            Ok(())
759        };
760
761        write_f32("erb_feat", erb_feat)?;
762        write_cplx("spec_feat_cplx", spec_feat)?;
763        write_f32("mask", mask)?;
764
765        // df_coefs shape: [NB_DF, DF_ORDER, 2]
766        let mut df_lines = String::new();
767        for f in 0..NB_DF {
768            for o in 0..DF_ORDER {
769                let re = df_coefs[[f, o, 0]];
770                let im = df_coefs[[f, o, 1]];
771                df_lines.push_str(&format!("{},{},{},{}\n", f, o, re, im));
772            }
773        }
774        let df_path = out_dir.join(format!("df_coefs_frame_{}.txt", frame_idx));
775        std::fs::write(df_path, df_lines)?;
776
777        Ok(())
778    }

781        &self,
782        name: &str,
783        frame_idx: usize,
784        shape: Vec<usize>,
785        data: &[f32],
786    ) -> Result<()> {
787        let out_dir = std::path::Path::new(env!("CARGO_MANIFEST_DIR"))
788            .join("debug_dump");
789        std::fs::create_dir_all(&out_dir)?;
790        let path = out_dir.join(format!("{}_frame_{}.txt", name, frame_idx));
791        let mut s = String::new();
792        s.push_str(&format!("# shape: {:?}\n", shape));
793        for (i, v) in data.iter().enumerate() {
794            s.push_str(&format!("{},{}\n", i, v));
795        }
796        std::fs::write(path, s)?;
797        Ok(())
798    }

800    fn apply_deep_filter(&mut self, spec: &mut [Complex32], df_coefs: &Array3<f32>) {
801        // Buffer contains the last DF_ORDER frames of ORIGINAL spectrum (before mask)
802        // Iterate from oldest to newest, multiply by coefs order 0..DF_ORDER-1
803        debug_assert!(self.rolling_spec_buf_x.len() >= DF_ORDER);
804
805        for freq in 0..NB_DF {
806            let mut filtered = Complex32::new(0.0, 0.0);
807
808            for (order, frame) in self.rolling_spec_buf_x.iter().take(DF_ORDER).enumerate() {
809                let orig_spec = frame[freq];
810                let coef = Complex32::new(df_coefs[[freq, order, 0]], df_coefs[[freq, order, 1]]);
811                filtered += orig_spec * coef;
812            }
813
814            // Replace only the DF frequency bins with filtered output
815            spec[freq] = filtered;
816        }
817    }

820/// Streaming wrapper with buffering
821pub struct DeepFilterStream {
822    processor: DeepFilterProcessor,
823    input_buffer: Vec<f32>,
824}
825
826impl DeepFilterStream {
827    pub fn new(model_dir: &Path) -> Result<Self> {
828        Ok(Self {
829            processor: DeepFilterProcessor::new(model_dir)?,
830            input_buffer: Vec::with_capacity(HOP_SIZE * 2),
831        })
832    }
833
834    /// Create stream with explicit inference mode.
835    pub fn new_with_mode(model_dir: &Path, mode: InferenceMode) -> Result<Self> {
836        Ok(Self {
837            processor: DeepFilterProcessor::new_with_mode(model_dir, mode)?,
838            input_buffer: Vec::with_capacity(HOP_SIZE * 2),
839        })
840    }
841
842    /// Create stream with explicit thread count
843    ///
844    /// # Arguments
845    /// * `model_dir` - Path to directory containing ONNX models
846    /// * `intra_threads` - Number of threads for intra-op parallelism
847    pub fn with_threads(model_dir: &Path, intra_threads: usize) -> Result<Self> {
848        let variant = ModelVariant::from_config(model_dir).unwrap_or(ModelVariant::DeepFilterNet3);
849        Ok(Self {
850            processor: DeepFilterProcessor::with_variant_and_threads(
851                model_dir,
852                variant,
853                Some(intra_threads),
854            )?,
855            input_buffer: Vec::with_capacity(HOP_SIZE * 2),
856        })
857    }
858
859    /// Perform warm-up inference to avoid cold-start latency
860    pub fn warmup(&mut self) -> Result<()> {
861        self.processor.warmup()
862    }
863
864    /// Process variable-length input
865    pub fn process(&mut self, input: &[f32]) -> Result<Vec<f32>> {
866        self.input_buffer.extend_from_slice(input);
867        let mut output = Vec::new();
868
869        while self.input_buffer.len() >= HOP_SIZE {
870            let frame_in: Vec<f32> = self.input_buffer.drain(..HOP_SIZE).collect();
871            let mut frame_out = vec![0.0f32; HOP_SIZE];
872            self.processor.process_frame(&frame_in, &mut frame_out)?;
873            output.extend_from_slice(&frame_out);
874        }
875
876        Ok(output)
877    }
878
879    pub fn flush(&mut self) -> Result<Vec<f32>> {
880        if self.input_buffer.is_empty() {
881            return Ok(Vec::new());
882        }
883        let valid_len = self.input_buffer.len();
884        self.input_buffer.resize(HOP_SIZE, 0.0);
885        let frame_in: Vec<f32> = self.input_buffer.drain(..).collect();
886        let mut frame_out = vec![0.0f32; HOP_SIZE];
887        self.processor.process_frame(&frame_in, &mut frame_out)?;
888        Ok(frame_out[..valid_len].to_vec())
889    }
890
891    pub fn reset(&mut self) {
892        self.processor.reset();
893        self.input_buffer.clear();
894    }
895
896    pub fn variant(&self) -> ModelVariant { self.processor.variant() }
897    pub fn frame_size(&self) -> usize { HOP_SIZE }
898    pub fn sample_rate(&self) -> usize { SAMPLE_RATE }
899
900    /// Latency in ms (10ms for LL variants, 30ms for standard)
901    pub fn latency_ms(&self) -> f32 {
902        let base = (FFT_SIZE - HOP_SIZE) as f32 / SAMPLE_RATE as f32 * 1000.0;
903        if self.processor.variant.is_low_latency() {
904            base // 10ms
905        } else {
906            base + 20.0 // +2 frames lookahead = 30ms total
907        }
908    }
909
910    /// Get mutable access to the underlying processor
911    pub fn processor_mut(&mut self) -> &mut DeepFilterProcessor {
912        &mut self.processor
913    }
914}

919fn _assert_processor_is_send() {
920    _assert_send::<DeepFilterProcessor>();
921    _assert_send::<DeepFilterStream>();
